2. event_competition — new columns

ALTER TABLE event_competition
ALTER TABLE public.event_competition
    ADD COLUMN IF NOT EXISTS discipline_code character varying(32),
    ADD COLUMN IF NOT EXISTS min_age integer,
    ADD COLUMN IF NOT EXISTS max_age integer,
    ADD COLUMN IF NOT EXISTS min_grade integer,
    ADD COLUMN IF NOT EXISTS max_grade integer,
    ADD COLUMN IF NOT EXISTS max_placements integer DEFAULT 3 NOT NULL,
    ADD COLUMN IF NOT EXISTS registration_deadline timestamp without time zone,
    ADD COLUMN IF NOT EXISTS rules_markdown text,
    ADD COLUMN IF NOT EXISTS requires_team_name boolean DEFAULT false NOT NULL;

ALTER TABLE public.event_competition
    ADD CONSTRAINT chk_event_competition__discipline
        CHECK (discipline_code IS NULL OR discipline_code IN (
            'SONG', 'SPEECH', 'DANCE', 'MUSIC', 'SPORTS', 'ART', 'OTHER'
        ));

ALTER TABLE public.event_competition
    ADD CONSTRAINT chk_event_competition__age_bounds
        CHECK (min_age IS NULL OR max_age IS NULL OR min_age <= max_age);

ALTER TABLE public.event_competition
    ADD CONSTRAINT chk_event_competition__grade_bounds
        CHECK (min_grade IS NULL OR max_grade IS NULL OR min_grade <= max_grade);

ALTER TABLE public.event_competition
    ADD CONSTRAINT chk_event_competition__max_placements
        CHECK (max_placements >= 1 AND max_placements <= 20);

COMMENT ON COLUMN public.event_competition.discipline_code IS
    'Normalized competition discipline for catalog filtering and icons.';
COMMENT ON COLUMN public.event_competition.rules_markdown IS
    'Per-competition regulations shown on public detail page.';
COMMENT ON COLUMN public.event_competition.registration_deadline IS
    'Optional per-competition cutoff; falls back to event_competition_settings.registration_deadline.';

3. event_competition_registration — team fields

ALTER TABLE public.event_competition_registration
    ADD COLUMN IF NOT EXISTS team_name character varying(200),
    ADD COLUMN IF NOT EXISTS team_display_name character varying(200);

COMMENT ON COLUMN public.event_competition_registration.team_name IS
    'Internal team label for GROUP leader registration (captain-entered).';
COMMENT ON COLUMN public.event_competition_registration.team_display_name IS
    'Public label on results board; defaults to team_name when null.';

4. event_competition_settings — extended scoring

ALTER TABLE public.event_competition_settings
    ADD COLUMN IF NOT EXISTS points_fourth integer DEFAULT 0 NOT NULL,
    ADD COLUMN IF NOT EXISTS default_max_placements integer DEFAULT 3 NOT NULL;

ALTER TABLE public.event_competition_settings
    ADD CONSTRAINT chk_event_competition_settings__default_placements
        CHECK (default_max_placements >= 1 AND default_max_placements <= 20);

When admin creates a new competition, frontend/backend pre-fills max_placements from default_max_placements. Champion rollup continues to use points_first … points_third plus points_fourth when enabled.

5. event_competition_group_member — activate & extend

v2.0 table exists — v2.1 adds roles and ordering
ALTER TABLE public.event_competition_group_member
    ADD COLUMN IF NOT EXISTS member_role character varying(20) DEFAULT 'MEMBER' NOT NULL,
    ADD COLUMN IF NOT EXISTS sort_order integer DEFAULT 0 NOT NULL,
    ADD COLUMN IF NOT EXISTS updated_at timestamp without time zone DEFAULT now() NOT NULL;

ALTER TABLE public.event_competition_group_member
    ADD CONSTRAINT chk_event_comp_group_member__role
        CHECK (member_role IN ('CAPTAIN', 'MEMBER'));

CREATE UNIQUE INDEX IF NOT EXISTS ux_event_comp_group_member__reg_participant
    ON public.event_competition_group_member(registration_id, participant_profile_id);

CREATE INDEX IF NOT EXISTS idx_event_comp_group_member__registration
    ON public.event_competition_group_member(registration_id);
