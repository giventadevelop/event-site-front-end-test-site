Spring Boot Implementation

Critical Requirements

  • Payment Orchestration: Route donation-based events to GiveButter based on event configuration
  • Data Persistence: Persist all donation transactions to database for admin dashboard statistics
  • QR Code Generation: Generate QR codes for donations (similar to ticket purchase flow)
  • Email Sending: Send confirmation emails via API endpoints (called from frontend)
  • Status Polling Endpoints: Provide endpoints for frontend to poll GiveButter donation status
  • Transaction Creation: Create donation transaction records via API endpoints (called from frontend after detecting success)
  • NO Webhook Handling: Frontend handles all webhook-like behavior through polling (similar to Stripe checkout flow)

1. Executive Summary

This PRD outlines the backend implementation of GiveButter integration for non-profit fundraiser events. The implementation extends the existing domain-agnostic payment infrastructure to support GiveButter as a payment provider, with automatic routing for donation-based events.

Key Objectives

  • Integrate GiveButter API as a payment provider adapter
  • Route donation-based events to GiveButter automatically
  • Persist donation transactions to database
  • Generate QR codes for donations
  • Send confirmation emails via API endpoints (called from frontend)
  • Provide status polling endpoints for frontend to check GiveButter donation status
  • Support ticketed fundraiser events with GiveButter routing
  • Create donation transactions via API endpoints (called from frontend after detecting success)

3.2 Payment Initialization

@Override public PaymentInitializeResponse initialize(PaymentInitializeRequest request) { try { // Get GiveButter API key from payment provider config PaymentProviderConfig config = getGiveButterConfig(request.getTenantId()); String apiKey = decryptApiKey(config.getConfigData()); // Build GiveButter donation request GiveButterDonationRequest gbRequest = new GiveButterDonationRequest(); gbRequest.setAmount(request.getAmount()); gbRequest.setEmail(request.getEmail()); gbRequest.setFirstName(request.getFirstName()); gbRequest.setLastName(request.getLastName()); gbRequest.setPhone(request.getPhone()); // Get campaign ID from event metadata or provider config String campaignId = getCampaignId(request.getEventId(), config); gbRequest.setCampaignId(campaignId); // Set return URLs gbRequest.setReturnUrl(request.getReturnUrl()); gbRequest.setCancelUrl(request.getCancelUrl()); // Add custom fields (prayer intention, etc.) if (request.getMetadata() != null) { Map<String, Object> metadata = request.getMetadata(); if (metadata.containsKey("prayerIntention")) { gbRequest.addCustomField("prayer_intention", metadata.get("prayerIntention").toString()); } if (metadata.containsKey("isRecurring")) { gbRequest.setRecurring((Boolean) metadata.get("isRecurring")); } if (metadata.containsKey("isAnonymous")) { gbRequest.setAnonymous((Boolean) metadata.get("isAnonymous")); } } // Call GiveButter API to create donation GiveButterDonationResponse gbResponse = apiClient.createDonation(apiKey, gbRequest); // Create payment transaction record PaymentTransaction paymentTransaction = createPaymentTransaction(request, gbResponse); // Create donation transaction record (if not ticketed) if (request.getPaymentType() == PaymentType.DONATION || request.getPaymentType() == PaymentType.DONATION_ZERO_FEE) { createDonationTransaction(request, paymentTransaction, gbResponse); } // Return checkout URL PaymentInitializeResponse response = new PaymentInitializeResponse(); response.setTransactionId(paymentTransaction.getId()); response.setCheckoutUrl(gbResponse.getCheckoutUrl()); response.setStatus("PENDING"); return response; } catch (Exception e) { throw new PaymentException("Failed to initialize GiveButter donation", e); } } private String getCampaignId(Long eventId, PaymentProviderConfig config) { // Priority 1: Event-specific campaign ID from metadata if (eventId != null) { EventDetails event = eventRepository.findById(eventId).orElse(null); if (event != null && event.getMetadata() != null) { Map<String, Object> metadata = parseMetadata(event.getMetadata()); if (metadata != null) { Map<String, Object> donationConfig = (Map<String, Object>) metadata.get("donationConfig"); if (donationConfig != null) { String campaignId = (String) donationConfig.get("givebutterCampaignId"); if (campaignId != null && !campaignId.trim().isEmpty()) { return campaignId; } } } } } // Priority 2: Provider config default campaign ID if (config.getMetadata() != null) { Map<String, Object> meta = config.getMetadata(); String defaultCampaignId = (String) meta.get("campaignId"); if (defaultCampaignId != null && !defaultCampaignId.trim().isEmpty()) { return defaultCampaignId; } } // Priority 3: Throw exception if no campaign ID found throw new PaymentException("GiveButter campaign ID not configured"); }

3.3 GiveButter API Client

// File: com.yourpackage.payment.adapter.givebutter.GiveButterApiClient.java @Service public class GiveButterApiClient { private static final String GIVEBUTTER_API_BASE = "https://api.givebutter.com/v1"; private final RestTemplate restTemplate; public GiveButterDonationResponse createDonation(String apiKey, GiveButterDonationRequest request) { String url = GIVEBUTTER_API_BASE + "/donations"; HttpHeaders headers = new HttpHeaders(); headers.set("Authorization", "Bearer " + apiKey); headers.setContentType(MediaType.APPLICATION_JSON); HttpEntity<GiveButterDonationRequest> entity = new HttpEntity<>(request, headers); ResponseEntity<GiveButterDonationResponse> response = restTemplate.postForEntity(url, entity, GiveButterDonationResponse.class); if (!response.getStatusCode().is2xxSuccessful()) { throw new PaymentException("GiveButter API error: " + response.getStatusCode()); } return response.getBody(); } public GiveButterDonationResponse getDonation(String apiKey, String donationId) { String url = GIVEBUTTER_API_BASE + "/donations/" + donationId; HttpHeaders headers = new HttpHeaders(); headers.set("Authorization", "Bearer " + apiKey); HttpEntity<Void> entity = new HttpEntity<>(headers); ResponseEntity<GiveButterDonationResponse> response = restTemplate.exchange(url, HttpMethod.GET, entity, GiveButterDonationResponse.class); if (!response.getStatusCode().is2xxSuccessful()) { throw new PaymentException("GiveButter API error: " + response.getStatusCode()); } return response.getBody(); } }

4. Payment Orchestration Service

4.1 Routing Logic

Update PaymentOrchestrationService to route donation-based events to GiveButter:

// File: com.yourpackage.payment.service.PaymentOrchestrationService.java private ProviderType determineProvider(PaymentInitializeRequest request) { // For OFFERING payment type, prefer Givebutter if configured if (PaymentType.OFFERING.equals(request.getPaymentType())) { boolean hasGivebutter = configRepository.existsByTenantIdAndProviderType( request.getTenantId(), ProviderType.GIVEBUTTER); if (hasGivebutter) { return ProviderType.GIVEBUTTER; } } // For TICKET_SALE payment type, check if event is configured as fundraiser if (PaymentType.TICKET_SALE.equals(request.getPaymentType())) { if (request.getEventId() != null) { EventDetails event = eventRepository.findById(request.getEventId()).orElse(null); if (event != null && event.getMetadata() != null) { Map<String, Object> metadata = parseMetadata(event.getMetadata()); if (metadata != null) { Boolean isFundraiserEvent = (Boolean) metadata.get("isFundraiserEvent"); Boolean isCharityEvent = (Boolean) metadata.get("isCharityEvent"); Map<String, Object> donationConfig = (Map<String, Object>) metadata.get("donationConfig"); if ((Boolean.TRUE.equals(isFundraiserEvent) || Boolean.TRUE.equals(isCharityEvent)) && donationConfig != null) { Boolean useZeroFeeProvider = (Boolean) donationConfig.get("useZeroFeeProvider"); String zeroFeeProvider = (String) donationConfig.get("zeroFeeProvider"); if (Boolean.TRUE.equals(useZeroFeeProvider) && "GIVEBUTTER".equals(zeroFeeProvider)) { boolean hasGivebutter = configRepository .existsByTenantIdAndProviderType( request.getTenantId(), ProviderType.GIVEBUTTER); if (hasGivebutter) { // Route ticketed fundraiser event to Givebutter return ProviderType.GIVEBUTTER; } } } } } } } // For DONATION payment type, check if zero-fee provider is configured if (PaymentType.DONATION.equals(request.getPaymentType()) || PaymentType.DONATION_ZERO_FEE.equals(request.getPaymentType())) { if (request.getEventId() != null) { EventDetails event = eventRepository.findById(request.getEventId()).orElse(null); if (event != null && event.getMetadata() != null) { Map<String, Object> metadata = parseMetadata(event.getMetadata()); if (metadata != null) { Map<String, Object> donationConfig = (Map<String, Object>) metadata.get("donationConfig"); if (donationConfig != null) { Boolean useZeroFeeProvider = (Boolean) donationConfig.get("useZeroFeeProvider"); String zeroFeeProvider = (String) donationConfig.get("zeroFeeProvider"); if (Boolean.TRUE.equals(useZeroFeeProvider) && "GIVEBUTTER".equals(zeroFeeProvider)) { boolean hasGivebutter = configRepository .existsByTenantIdAndProviderType( request.getTenantId(), ProviderType.GIVEBUTTER); if (hasGivebutter) { return ProviderType.GIVEBUTTER; } } } } } } } // Default to Stripe (primary provider for regular tickets) return ProviderType.STRIPE; } /** * Parse metadata JSON string from TEXT column. * Metadata is stored as TEXT (not JSONB) matching existing codebase pattern. */ private Map<String, Object> parseMetadata(String metadataJson) { if (metadataJson == null || metadataJson.trim().isEmpty()) { return null; } try { ObjectMapper mapper = new ObjectMapper(); return mapper.readValue(metadataJson, Map.class); } catch (Exception e) { log.error("Failed to parse event metadata JSON", e); return null; } }

5. Donation Transaction Service

5.1 Create Donation Transaction

// File: com.yourpackage.donation.service.DonationTransactionService.java @Service public class DonationTransactionService { @Autowired private DonationTransactionRepository donationTransactionRepository; @Autowired private QrCodeService qrCodeService; @Autowired private EmailService emailService; public DonationTransaction createDonationTransaction( PaymentInitializeRequest request, PaymentTransaction paymentTransaction, GiveButterDonationResponse gbResponse) { DonationTransaction donation = new DonationTransaction(); donation.setTenantId(request.getTenantId()); donation.setEventId(request.getEventId()); donation.setPaymentTransactionId(paymentTransaction.getId()); donation.setTransactionReference("GB-" + paymentTransaction.getId()); donation.setGivebutterDonationId(gbResponse.getDonationId()); donation.setAmount(request.getAmount()); donation.setEmail(request.getEmail()); donation.setFirstName(request.getFirstName()); donation.setLastName(request.getLastName()); donation.setPhone(request.getPhone()); donation.setStatus("PENDING"); // Extract metadata if (request.getMetadata() != null) { Map<String, Object> metadata = request.getMetadata(); if (metadata.containsKey("prayerIntention")) { donation.setPrayerIntention( metadata.get("prayerIntention").toString()); } if (metadata.containsKey("isRecurring")) { donation.setIsRecurring( (Boolean) metadata.get("isRecurring")); } if (metadata.containsKey("isAnonymous")) { donation.setIsAnonymous( (Boolean) metadata.get("isAnonymous")); } } donation = donationTransactionRepository.save(donation); return donation; } }

6. Status Polling Endpoints (No Webhook Handling)

CRITICAL: Following the existing checkout flow pattern, the frontend handles all webhook-like behavior through polling. The backend provides status endpoints for the frontend to poll, but does NOT implement webhook listeners.

6.1 GiveButter Donation Status Endpoint

Provide an endpoint for the frontend to poll GiveButter donation status:

// File: com.yourpackage.controller.GiveButterStatusController.java @RestController @RequestMapping("/api/proxy/givebutter") public class GiveButterStatusController { @Autowired private GiveButterApiClient apiClient; @Autowired private PaymentProviderConfigRepository configRepository; @GetMapping("/donations/{donationId}/status") public ResponseEntity<GiveButterDonationStatusDTO> getDonationStatus( @PathVariable String donationId, @RequestParam(required = false) Long tenantId) { try { // Get GiveButter config PaymentProviderConfig config = getGiveButterConfig(tenantId); String apiKey = decryptApiKey(config.getConfigData()); // Call GiveButter API to get donation status GiveButterDonationResponse donation = apiClient.getDonation(apiKey, donationId); GiveButterDonationStatusDTO statusDTO = new GiveButterDonationStatusDTO(); statusDTO.setDonationId(donationId); statusDTO.setStatus(mapGiveButterStatus(donation.getStatus())); statusDTO.setAmount(donation.getAmount()); statusDTO.setEmail(donation.getEmail()); statusDTO.setCreatedAt(donation.getCreatedAt()); return ResponseEntity.ok(statusDTO); } catch (Exception e) { log.error("Error fetching GiveButter donation status", e); return ResponseEntity.status(500).build(); } } private String mapGiveButterStatus(String gbStatus) { // Map GiveButter status to our status enum switch (gbStatus.toLowerCase()) { case "completed": case "succeeded": return "COMPLETED"; case "pending": return "PENDING"; case "failed": return "FAILED"; case "cancelled": return "CANCELLED"; default: return "PENDING"; } } }

// File: com.yourpackage.controller.DonationTransactionController.java @RestController @RequestMapping("/api/donation-transactions") public class DonationTransactionController { @Autowired private DonationTransactionService donationService; @GetMapping("/{id}") public ResponseEntity<DonationTransactionDTO> getDonation( @PathVariable Long id, @RequestParam(required = false) Long eventId) { DonationTransaction donation = donationService.findById(id); // Verify event ID matches if provided if (eventId != null && !eventId.equals(donation.getEventId())) { return ResponseEntity.badRequest().build(); } return ResponseEntity.ok(convertToDTO(donation)); } @PostMapping("/events/{eventId}/donations/{transactionId}/qrcode") public ResponseEntity<Map<String, String>> generateQrCode( @PathVariable Long eventId, @PathVariable Long transactionId) { DonationTransaction donation = donationService.findById(transactionId); if (!eventId.equals(donation.getEventId())) { return ResponseEntity.badRequest().build(); } String qrCodeUrl = donationService.generateQrCode(eventId, transactionId); Map<String, String> response = new HashMap<>(); response.put("qrCodeUrl", qrCodeUrl); return ResponseEntity.ok(response); } @PostMapping("/events/{eventId}/donations/{transactionId}/send-email") public ResponseEntity<Map<String, Boolean>> sendEmail( @PathVariable Long eventId, @PathVariable Long transactionId, @RequestParam String to) { DonationTransaction donation = donationService.findById(transactionId); if (!eventId.equals(donation.getEventId())) { return ResponseEntity.badRequest().build(); } boolean sent = donationService.sendConfirmationEmail(eventId, transactionId, to); Map<String, Boolean> response = new HashMap<>(); response.put("emailSent", sent); return ResponseEntity.ok(response); } }

  • Status Polling Endpoints: Backend provides endpoints for frontend to poll donation status
  • Transaction Creation Endpoints: Backend provides endpoints for frontend to create transactions after detecting success

  • Pattern: Similar to Stripe checkout flow where frontend polls and creates transactions
