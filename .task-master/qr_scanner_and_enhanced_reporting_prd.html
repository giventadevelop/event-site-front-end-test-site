1. Executive Summary

This PRD outlines the requirements for implementing four critical enhancements to the event ticketing system:

  1. Web-Based QR Scanner - Upgrade from manual check-in to camera-based scanning
  2. Camera-Based Scanning - Enable real-time QR code scanning using device cameras
  3. Check-In History & Analytics - Enhance basic check-in tracking with comprehensive analytics
  4. Sales Reports - Expand basic reporting to comprehensive sales analytics
Business Value: These enhancements will transform the event check-in process from manual to automated, provide real-time insights into event attendance, and enable data-driven decision making for event organizers.

4. Functional Requirements

4.1 Web-Based QR Scanner HIGH PRIORITY

FR-1.1: Camera Access & QR Scanning

  • System must request camera permissions from user
  • System must support both front-facing (mobile) and rear-facing (desktop webcam) cameras
  • System must allow users to switch between cameras
  • QR scanner must continuously scan for QR codes
  • QR scanner must provide visual feedback (scanning overlay, success/error indicators)
  • QR scanner must work on both desktop (webcam) and mobile (camera) devices

FR-1.2: QR Code Validation

  • System must validate QR code format (transaction ID, event ID, or URL)
  • System must verify ticket exists in database
  • System must check ticket status (not REFUNDED, not CANCELLED)
  • System must verify ticket belongs to correct event
  • System must check if ticket is already checked in (with option to re-check-in)
  • System must display clear validation messages (valid/invalid ticket)

FR-1.3: Check-In Processing

  • Upon valid QR scan, system must automatically navigate to check-in page
  • System must pre-populate check-in form with ticket details
  • System must allow admin to adjust number of guests checking in
  • System must record check-in timestamp
  • System must update checkInStatus to 'CHECKED_IN'
  • System must update checkInTime with current timestamp
  • System must update numberOfGuestsCheckedIn
  • System must provide success confirmation after check-in

FR-1.4: Error Handling

  • System must handle camera permission denial gracefully
  • System must handle invalid QR codes with clear error messages
  • System must handle network errors during validation
  • System must handle duplicate check-ins (show warning, allow override)
  • System must handle expired or invalid tickets

4.2 Camera-Based Scanning HIGH PRIORITY

FR-2.1: Browser Camera Integration

  • System must use browser MediaDevices API for camera access
  • System must support getUserMedia() for webcam/camera access
  • System must handle camera initialization errors
  • System must provide fallback if camera is unavailable
  • System must optimize camera performance (frame rate, resolution)

FR-2.2: QR Code Detection

  • System must use QR code detection library (react-qr-reader or @zxing/library)
  • System must detect QR codes in real-time from camera feed
  • System must handle multiple QR codes in frame (select first valid)
  • System must provide visual scanning overlay (targeting box)
  • System must provide audio/visual feedback on successful scan

FR-2.3: Mobile Optimization

  • System must work on iOS Safari (with proper permissions)
  • System must work on Android Chrome
  • System must handle mobile camera orientation changes
  • System must optimize for mobile performance (battery, processing)
  • System must provide touch-friendly controls

4.3 Check-In History & Analytics MEDIUM PRIORITY

8.1 QR Verification API Route

// src/app/api/qr/verify/route.ts import { NextRequest, NextResponse } from 'next/server'; import { fetchWithJwtRetry } from '@/lib/proxyHandler'; export async function GET(req: NextRequest) { try { const { searchParams } = new URL(req.url); const qrData = searchParams.get('data'); if (!qrData) { return NextResponse.json( { valid: false, error: 'No QR data provided' }, { status: 400 } ); } // Extract transaction ID from QR data const transactionId = extractTransactionId(qrData); if (!transactionId) { return NextResponse.json( { valid: false, error: 'Invalid QR code format' }, { status: 400 } ); } // Fetch transaction from backend const API_BASE_URL = process.env.NEXT_PUBLIC_API_BASE_URL; const response = await fetchWithJwtRetry( `${API_BASE_URL}/api/event-ticket-transactions/${transactionId}`, { cache: 'no-store' } ); if (!response.ok) { return NextResponse.json( { valid: false, error: 'Ticket not found' }, { status: 404 } ); } const transaction = await response.json(); // Validate ticket const validation = validateTicket(transaction); return NextResponse.json({ valid: validation.isValid, transaction: validation.isValid ? transaction : null, message: validation.message, error: validation.isValid ? null : validation.error }); } catch (error) { console.error('QR Verification Error:', error); return NextResponse.json( { valid: false, error: 'Verification failed' }, { status: 500 } ); } } function extractTransactionId(qrData: string): string | null { // Handle URL format: /admin/qrcode-scan/.../transactions/123 const urlMatch = qrData.match(/transactions\/(\d+)/); if (urlMatch) return urlMatch[1]; // Handle JSON format: {"transactionId": 123, "eventId": 456} try { const parsed = JSON.parse(qrData); return parsed.transactionId || parsed.id || null; } catch { // Handle direct ID return /^\d+$/.test(qrData) ? qrData : null; } } function validateTicket(transaction: any): { isValid: boolean; message: string; error?: string } { if (!transaction) { return { isValid: false, message: 'Invalid ticket', error: 'Transaction not found' }; } if (transaction.status === 'REFUNDED') { return { isValid: false, message: 'Invalid ticket', error: 'Ticket has been refunded' }; } if (transaction.status === 'CANCELLED') { return { isValid: false, message: 'Invalid ticket', error: 'Ticket has been cancelled' }; } return { isValid: true, message: 'Ticket valid - Access granted' }; }

8.2 Check-In Analytics Server Actions

// src/app/admin/events/[id]/check-in-analytics/ApiServerActions.ts "use server"; import { fetchWithJwtRetry } from '@/lib/proxyHandler'; import type { EventTicketTransactionDTO } from '@/types'; const API_BASE_URL = process.env.NEXT_PUBLIC_API_BASE_URL; export async function fetchCheckInHistoryServer( eventId: string, options: { startDate?: string; endDate?: string; page?: number; size?: number; } = {} ): Promise<{ transactions: EventTicketTransactionDTO[]; totalCount: number }> { const params = new URLSearchParams({ 'eventId.equals': eventId, 'checkInStatus.equals': 'CHECKED_IN', 'sort': 'checkInTime,desc', page: (options.page || 0).toString(), size: (options.size || 20).toString(), }); if (options.startDate) { params.append('checkInTime.greaterThanOrEqual', options.startDate); } if (options.endDate) { params.append('checkInTime.lessThanOrEqual', options.endDate); } const url = `${API_BASE_URL}/api/event-ticket-transactions?${params.toString()}`; const response = await fetchWithJwtRetry(url, { cache: 'no-store' }); if (!response.ok) { throw new Error('Failed to fetch check-in history'); } const transactions = await response.json(); const totalCount = parseInt(response.headers.get('x-total-count') || '0', 10); return { transactions: Array.isArray(transactions) ? transactions : [], totalCount }; } export async function fetchCheckInAnalyticsServer(eventId: string): Promise { // Fetch all checked-in transactions const params = new URLSearchParams({ 'eventId.equals': eventId, 'checkInStatus.equals': 'CHECKED_IN', 'size': '1000', // Get all for analytics }); const url = `${API_BASE_URL}/api/event-ticket-transactions?${params.toString()}`; const response = await fetchWithJwtRetry(url, { cache: 'no-store' }); if (!response.ok) { throw new Error('Failed to fetch check-in analytics'); } const transactions = await response.json(); const allTransactions = Array.isArray(transactions) ? transactions : []; // Calculate analytics const totalTickets = allTransactions.reduce((sum, t) => sum + (t.quantity || 0), 0); const checkedInCount = allTransactions.length; const totalGuestsCheckedIn = allTransactions.reduce((sum, t) => sum + (t.numberOfGuestsCheckedIn || 0), 0); // Group by hour const checkInsByHour: Record<string, number> = {}; allTransactions.forEach(t => { if (t.checkInTime) { const hour = new Date(t.checkInTime).toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' }); checkInsByHour[hour] = (checkInsByHour[hour] || 0) + 1; } }); return { eventId: parseInt(eventId, 10), totalTickets, checkedInCount, totalGuestsCheckedIn, checkInPercentage: totalTickets > 0 ? (checkedInCount / totalTickets) * 100 : 0, checkInsByHour: Object.entries(checkInsByHour).map(([hour, count]) => ({ hour, count })), }; }
