// ✅ ADD new result class to hold fee, tax, and net payout public static class StripeFeeTaxNetResult { private final BigDecimal fee; private final BigDecimal tax; private final BigDecimal netPayout; public StripeFeeTaxNetResult(BigDecimal fee, BigDecimal tax, BigDecimal netPayout) { this.fee = fee; this.tax = tax; this.netPayout = netPayout; } public BigDecimal getFee() { return fee; } public BigDecimal getTax() { return tax; } public BigDecimal getNetPayout() { return netPayout; } } // ✅ ADD new method to retrieve fee, tax, and net payout together /** * Retrieve Stripe fee, tax, and net payout amount for a payment intent. * This method retrieves all three values in a single call for efficiency. * * @param tenantId The tenant ID * @param paymentIntentId The Stripe payment intent ID * @param checkoutSessionId The Stripe checkout session ID (optional, for tax retrieval) * @return StripeFeeTaxNetResult containing fee, tax, and net payout, or null if not found */ public StripeFeeTaxNetResult getStripeFeeTaxAndNet( String tenantId, String paymentIntentId, String checkoutSessionId ) { if (paymentIntentId == null || paymentIntentId.isEmpty()) { log.warn("Payment intent ID is null or empty for tenant: {}", tenantId); return new StripeFeeTaxNetResult(null, null, null); } String apiKey = getStripeApiKey(tenantId); if (apiKey == null || apiKey.isEmpty()) { log.warn("Stripe API key not found for tenant: {}", tenantId); return new StripeFeeTaxNetResult(null, null, null); } try { Stripe.apiKey = apiKey; // Step 1: Retrieve PaymentIntent log.debug("Retrieving PaymentIntent {} for tenant {}", paymentIntentId, tenantId); PaymentIntent paymentIntent = PaymentIntent.retrieve(paymentIntentId); // Step 2: Get charges for this PaymentIntent Map<String, Object> chargeParams = new java.util.HashMap<>(); chargeParams.put("payment_intent", paymentIntentId); chargeParams.put("limit", 1); List<Charge> charges = Charge.list(chargeParams).getData(); if (charges.isEmpty()) { log.warn("No charges found for PaymentIntent {} for tenant {}", paymentIntentId, tenantId); return new StripeFeeTaxNetResult(null, null, null); } Charge charge = charges.get(0); // Step 3: Get balance transaction (contains fee and net) if (charge.getBalanceTransaction() == null) { log.warn("Charge {} missing balance_transaction for tenant {}", charge.getId(), tenantId); return new StripeFeeTaxNetResult(null, null, null); } String balanceTransactionId = charge.getBalanceTransaction(); log.debug("Retrieving balance transaction {} for tenant {}", balanceTransactionId, tenantId); BalanceTransaction balanceTx = BalanceTransaction.retrieve(balanceTransactionId); // Step 4: Extract fee (convert from cents to dollars) Long feeInCents = balanceTx.getFee(); BigDecimal feeAmount = null; if (feeInCents != null) { feeAmount = BigDecimal.valueOf(feeInCents).divide(BigDecimal.valueOf(100), 2, RoundingMode.HALF_UP); } // NEW: Step 5: Extract net amount (convert from cents to dollars) Long netInCents = balanceTx.getNet(); BigDecimal netAmount = null; if (netInCents != null) { netAmount = BigDecimal.valueOf(netInCents).divide(BigDecimal.valueOf(100), 2, RoundingMode.HALF_UP); log.debug("Retrieved Stripe net payout {} for PaymentIntent {} for tenant {}", netAmount, paymentIntentId, tenantId); } // Step 6: Retrieve tax amount (existing logic) BigDecimal taxAmount = getStripeTax(tenantId, paymentIntentId, checkoutSessionId); log.debug("Retrieved Stripe fee={}, tax={}, net={} for PaymentIntent {} for tenant {}", feeAmount, taxAmount, netAmount, paymentIntentId, tenantId); return new StripeFeeTaxNetResult(feeAmount, taxAmount, netAmount); } catch (StripeException e) { log.error("Stripe API error retrieving fee/tax/net for PaymentIntent {} for tenant {}: {}", paymentIntentId, tenantId, e.getMessage()); return new StripeFeeTaxNetResult(null, null, null); } catch (Exception e) { log.error("Unexpected error retrieving Stripe fee/tax/net for PaymentIntent {} for tenant {}: {}", paymentIntentId, tenantId, e.getMessage(), e); return new StripeFeeTaxNetResult(null, null, null); } }
