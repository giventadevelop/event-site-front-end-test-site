1. Current State Analysis

1.1 Existing Event Type Differentiation

✅ Current Implementation:

  • Field: event_details.is_sports_event (boolean)
  • Purpose: Marks events as sports events
  • Type: Boolean (true/false)
  • Default: false
  • Location: event_details table, column is_sports_event

❌ Missing:

  • No field to mark events as arts competitions
  • No way to differentiate between:
    • Regular events
    • Sports competitions
    • Arts competitions

Recommended Approach:

Option 1: Add is_arts_competition Field (Recommended)
  • Add is_arts_competition boolean field to event_details table
  • Events can be:
    • Neither (regular event): is_sports_event = false, is_arts_competition = false
    • Sports: is_sports_event = true, is_arts_competition = false
    • Arts: is_sports_event = false, is_arts_competition = true
  • Simple, backward compatible, allows both flags if needed in future
Option 2: Use Event Type Classification
  • Leverage existing event_type_details table
  • Create event types: "Sports Competition", "Arts Competition"
  • Filter by event_type_id
  • More flexible but requires event type setup

2.3 New Arts Competition Results Table

Create event_arts_competition_result Table

-- ===================================================
-- Arts Competition Results
-- ===================================================

CREATE TABLE public.event_arts_competition_result (
    id bigint DEFAULT nextval('public.sequence_generator'::regclass) NOT NULL,
    tenant_id character varying(255) NOT NULL,
    event_id bigint NOT NULL,

    -- Competition Details
    category character varying(255) NOT NULL,
    subcategory character varying(255),
    competition_date date,

    -- Winner Information
    first_place_participant_name character varying(255),
    first_place_participant_details text,
    first_place_prize character varying(255),
    first_place_score numeric(10,2),

    second_place_participant_name character varying(255),
    second_place_participant_details text,
    second_place_prize character varying(255),
    second_place_score numeric(10,2),

    third_place_participant_name character varying(255),
    third_place_participant_details text,
    third_place_prize character varying(255),
    third_place_score numeric(10,2),

    -- Additional Information
    total_participants integer,
    judges_names text,
    remarks text,
    result_announcement_date timestamp without time zone,

    -- Media
    winner_photo_url character varying(1024),
    ceremony_photo_url character varying(1024),

    -- Status
    is_published boolean DEFAULT false,
    results_visible_to character varying(50) DEFAULT 'ALL',

    -- Audit Fields
    created_at timestamp without time zone DEFAULT now() NOT NULL,
    updated_at timestamp without time zone DEFAULT now() NOT NULL,
    created_by_id bigint,

    -- Constraints
    CONSTRAINT event_arts_competition_result_pkey PRIMARY KEY (id),

    -- Unique constraint: one result per event category
    CONSTRAINT ux_arts_competition_result__tenant_event_category
        UNIQUE (tenant_id, event_id, category),

    -- Foreign Keys
    CONSTRAINT fk_arts_competition_result__event
        FOREIGN KEY (event_id)
        REFERENCES public.event_details(id) ON DELETE CASCADE,

    CONSTRAINT fk_arts_competition_result__created_by
        FOREIGN KEY (created_by_id)
        REFERENCES public.user_profile(id) ON DELETE SET NULL,

    -- Check Constraints
    CONSTRAINT check_total_participants_positive
        CHECK (total_participants IS NULL OR total_participants >= 0),

    CONSTRAINT check_results_visible_to
        CHECK (results_visible_to IN ('ALL', 'MEMBERS_ONLY', 'ADMIN_ONLY')),

    CONSTRAINT check_scores_valid
        CHECK (
            (first_place_score IS NULL OR first_place_score >= 0) AND
            (second_place_score IS NULL OR second_place_score >= 0) AND
            (third_place_score IS NULL OR third_place_score >= 0)
        )
);

-- Table Comment
COMMENT ON TABLE public.event_arts_competition_result IS
    'Results for arts competitions (dance, music, art, etc.) with winners and prizes';

-- Column Comments
COMMENT ON COLUMN public.event_arts_competition_result.category IS
    'Competition category (e.g., "Classical Dance", "Western Music", "Painting")';

COMMENT ON COLUMN public.event_arts_competition_result.subcategory IS
    'Subcategory if applicable (e.g., "Solo", "Group", "Under 18")';

COMMENT ON COLUMN public.event_arts_competition_result.results_visible_to IS
    'Who can view results: ALL (public), MEMBERS_ONLY, ADMIN_ONLY';

COMMENT ON COLUMN public.event_arts_competition_result.is_published IS
    'Whether results are published and visible to public';

-- Indexes
CREATE INDEX idx_arts_competition_result__tenant_id
    ON public.event_arts_competition_result(tenant_id);

CREATE INDEX idx_arts_competition_result__event_id
    ON public.event_arts_competition_result(event_id);

CREATE INDEX idx_arts_competition_result__category
    ON public.event_arts_competition_result(category);

CREATE INDEX idx_arts_competition_result__is_published
    ON public.event_arts_competition_result(is_published)
    WHERE is_published = true;

CREATE INDEX idx_arts_competition_result__competition_date
    ON public.event_arts_competition_result(competition_date);
